##### 1094. 拼车

差分 标记变化量 -->  求和最大值
 
```C++
class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        int n = trips.size();
        vector<int> nums(1002,0);
        for(auto & trip:trips){
            int l = trip[1],r = trip[2];
            int c = trip[0];
            nums[l] += c;
            nums[r] -= c;
        }

        int sum = 0;
        for(int i = 0;i < 1002;i++){
            sum += nums[i];
            if(sum > capacity) return false;
        }
        return true;
    }
};
```

#### 1109. 航班预订统计

差分模板题，数组索引从1开始的，要转换。

```C++
class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> nums(n,0);
        for(auto & book:bookings){
            int l = book[0] - 1,r = book[1] - 1;
            int c = book[2];
            nums[l] += c;
            if(r + 1 < n) nums[r + 1] -= c;
        }

        for(int i = 1;i < n;i++){
            nums[i] += nums[i - 1];
        }
        return nums;
    }
};
```
